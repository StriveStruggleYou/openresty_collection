 ```   
    local headers = ngx.resp.get_headers()
```  

### 2.根据cookie 内容进行判断
```  
  local ck = require "resty.cookie"
  local cookie, err = ck:new()
  if not cookie then
      ngx.log(ngx.ERR, err)
      return
  end
```

### 3.根据url参数 进行内容判断
```
  local arg = ngx.req.get_uri_args()
  for k,v in pairs(arg) do
      ngx.log(ngx.ERR, "url pairs: ", k.." ==> "..v)
  end
```
